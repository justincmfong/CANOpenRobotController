123 template<typename LhsScalar, typename RhsScalar, int KcFactor, typename Index>
124 void evaluateProductBlockingSizesHeuristic(Index& k, Index& m, Index& n, Index num_threads = 1)
125 {
126  typedef gebp_traits<LhsScalar,RhsScalar> Traits;
127 
128  // Explanations:
129  // Let's recall that the product algorithms form mc x kc vertical panels A' on the lhs and
130  // kc x nc blocks B' on the rhs. B' has to fit into L2/L3 cache. Moreover, A' is processed
131  // per mr x kc horizontal small panels where mr is the blocking size along the m dimension
132  // at the register level. This small horizontal panel has to stay within L1 cache.
133  std::ptrdiff_t l1, l2, l3;
134  manage_caching_sizes(GetAction, &l1, &l2, &l3);
135  #ifdef EIGEN_VECTORIZE_AVX512
136  // We need to find a rationale for that, but without this adjustment,
137  // performance with AVX512 is pretty bad, like -20% slower.
138  // One reason is that with increasing packet-size, the blocking size k
139  // has to become pretty small if we want that 1 lhs panel fit within L1.
140  // For instance, with the 3pX4 kernel and double, the size of the lhs+rhs panels are:
141  // k*(3*64 + 4*8) Bytes, with l1=32kBytes, and k%8=0, we have k=144.
142  // This is quite small for a good reuse of the accumulation registers.
143  l1 *= 4;
144  #endif
145 
146  if (num_threads > 1) {
147  typedef typename Traits::ResScalar ResScalar;
148  enum {
149  kdiv = KcFactor * (Traits::mr * sizeof(LhsScalar) + Traits::nr * sizeof(RhsScalar)),
150  ksub = Traits::mr * Traits::nr * sizeof(ResScalar),
151  kr = 8,
152  mr = Traits::mr,
153  nr = Traits::nr
154  };
155  // Increasing k gives us more time to prefetch the content of the "C"
156  // registers. However once the latency is hidden there is no point in
157  // increasing the value of k, so we'll cap it at 320 (value determined
158  // experimentally).
159  const Index k_cache = (numext::mini<Index>)((l1-ksub)/kdiv, 320);
160  if (k_cache < k) {
161  k = k_cache - (k_cache % kr);
162  eigen_internal_assert(k > 0);
163  }
164 
165  const Index n_cache = (l2-l1) / (nr * sizeof(RhsScalar) * k);
166  const Index n_per_thread = numext::div_ceil(n, num_threads);
167  if (n_cache <= n_per_thread) {
168  // Don't exceed the capacity of the l2 cache.
169  eigen_internal_assert(n_cache >= static_cast<Index>(nr));
170  n = n_cache - (n_cache % nr);
171  eigen_internal_assert(n > 0);
172  } else {
173  n = (numext::mini<Index>)(n, (n_per_thread + nr - 1) - ((n_per_thread + nr - 1) % nr));
174  }
175 
176  if (l3 > l2) {
177  // l3 is shared between all cores, so we'll give each thread its own chunk of l3.
178  const Index m_cache = (l3-l2) / (sizeof(LhsScalar) * k * num_threads);
179  const Index m_per_thread = numext::div_ceil(m, num_threads);
180  if(m_cache < m_per_thread && m_cache >= static_cast<Index>(mr)) {
181  m = m_cache - (m_cache % mr);
182  eigen_internal_assert(m > 0);
183  } else {
184  m = (numext::mini<Index>)(m, (m_per_thread + mr - 1) - ((m_per_thread + mr - 1) % mr));
185  }
186  }
187  }
188  else {
189  // In unit tests we do not want to use extra large matrices,
190  // so we reduce the cache size to check the blocking strategy is not flawed
191 #ifdef EIGEN_DEBUG_SMALL_PRODUCT_BLOCKS
192  l1 = 9*1024;
193  l2 = 32*1024;
194  l3 = 512*1024;
195 #endif
196 
197  // Early return for small problems because the computation below are time consuming for small problems.
198  // Perhaps it would make more sense to consider k*n*m??
199  // Note that for very tiny problem, this function should be bypassed anyway
200  // because we use the coefficient-based implementation for them.
201  if((numext::maxi)(k,(numext::maxi)(m,n))<48)
202  return;
203 
204  typedef typename Traits::ResScalar ResScalar;
205  enum {
206  k_peeling = 8,
207  k_div = KcFactor * (Traits::mr * sizeof(LhsScalar) + Traits::nr * sizeof(RhsScalar)),
208  k_sub = Traits::mr * Traits::nr * sizeof(ResScalar)
209  };
210 
211  // ---- 1st level of blocking on L1, yields kc ----
212 
213  // Blocking on the third dimension (i.e., k) is chosen so that an horizontal panel
214  // of size mr x kc of the lhs plus a vertical panel of kc x nr of the rhs both fits within L1 cache.
215  // We also include a register-level block of the result (mx x nr).
216  // (In an ideal world only the lhs panel would stay in L1)
217  // Moreover, kc has to be a multiple of 8 to be compatible with loop peeling, leading to a maximum blocking size of:
218  const Index max_kc = numext::maxi<Index>(((l1-k_sub)/k_div) & (~(k_peeling-1)),1);
219  const Index old_k = k;
220  if(k>max_kc)
221  {
222  // We are really blocking on the third dimension:
223  // -> reduce blocking size to make sure the last block is as large as possible
224  // while keeping the same number of sweeps over the result.
225  k = (k%max_kc)==0 ? max_kc
226  : max_kc - k_peeling * ((max_kc-1-(k%max_kc))/(k_peeling*(k/max_kc+1)));
227 
228  eigen_internal_assert(((old_k/k) == (old_k/max_kc)) && "the number of sweeps has to remain the same");
229  }
230 
231  // ---- 2nd level of blocking on max(L2,L3), yields nc ----
232 
233  // TODO find a reliable way to get the actual amount of cache per core to use for 2nd level blocking, that is:
234  // actual_l2 = max(l2, l3/nb_core_sharing_l3)
235  // The number below is quite conservative: it is better to underestimate the cache size rather than overestimating it)
236  // For instance, it corresponds to 6MB of L3 shared among 4 cores.
237  #ifdef EIGEN_DEBUG_SMALL_PRODUCT_BLOCKS
238  const Index actual_l2 = l3;
239  #else
240  const Index actual_l2 = 1572864; // == 1.5 MB
241  #endif
242 
243  // Here, nc is chosen such that a block of kc x nc of the rhs fit within half of L2.
244  // The second half is implicitly reserved to access the result and lhs coefficients.
245  // When k<max_kc, then nc can arbitrarily growth. In practice, it seems to be fruitful
246  // to limit this growth: we bound nc to growth by a factor x1.5.
247  // However, if the entire lhs block fit within L1, then we are not going to block on the rows at all,
248  // and it becomes fruitful to keep the packed rhs blocks in L1 if there is enough remaining space.
249  Index max_nc;
250  const Index lhs_bytes = m * k * sizeof(LhsScalar);
251  const Index remaining_l1 = l1- k_sub - lhs_bytes;
252  if(remaining_l1 >= Index(Traits::nr*sizeof(RhsScalar))*k)
253  {
254  // L1 blocking
255  max_nc = remaining_l1 / (k*sizeof(RhsScalar));
256  }
257  else
258  {
259  // L2 blocking
260  max_nc = (3*actual_l2)/(2*2*max_kc*sizeof(RhsScalar));
261  }
262  // WARNING Below, we assume that Traits::nr is a power of two.
263  Index nc = numext::mini<Index>(actual_l2/(2*k*sizeof(RhsScalar)), max_nc) & (~(Traits::nr-1));
264  if(n>nc)
265  {
266  // We are really blocking over the columns:
267  // -> reduce blocking size to make sure the last block is as large as possible
268  // while keeping the same number of sweeps over the packed lhs.
269  // Here we allow one more sweep if this gives us a perfect match, thus the commented "-1"
270  n = (n%nc)==0 ? nc
271  : (nc - Traits::nr * ((nc/*-1*/-(n%nc))/(Traits::nr*(n/nc+1))));
272  }
273  else if(old_k==k)
274  {
275  // So far, no blocking at all, i.e., kc==k, and nc==n.
276  // In this case, let's perform a blocking over the rows such that the packed lhs data is kept in cache L1/L2
277  // TODO: part of this blocking strategy is now implemented within the kernel itself, so the L1-based heuristic here should be obsolete.
278  Index problem_size = k*n*sizeof(LhsScalar);
279  Index actual_lm = actual_l2;
280  Index max_mc = m;
281  if(problem_size<=1024)
282  {
283  // problem is small enough to keep in L1
284  // Let's choose m such that lhs's block fit in 1/3 of L1
285  actual_lm = l1;
286  }
287  else if(l3!=0 && problem_size<=32768)
288  {
289  // we have both L2 and L3, and problem is small enough to be kept in L2
290  // Let's choose m such that lhs's block fit in 1/3 of L2
291  actual_lm = l2;
292  max_mc = (numext::mini<Index>)(576,max_mc);
293  }
294  Index mc = (numext::mini<Index>)(actual_lm/(3*k*sizeof(LhsScalar)), max_mc);
295  if (mc > Traits::mr) mc -= mc % Traits::mr;
296  else if (mc==0) return;
297  m = (m%mc)==0 ? mc
298  : (mc - Traits::mr * ((mc/*-1*/-(m%mc))/(Traits::mr*(m/mc+1))));
299  }
300  }
301 }
